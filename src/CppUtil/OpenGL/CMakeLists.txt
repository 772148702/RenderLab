GET_DIR_NAME(DIRNAME)

    add_definitions( -DUSE_QT_OPENGL_API )

	set(CMAKE_AUTOMOC ON)

	set(CMAKE_AUTOMOC OFF)
	set(CMAKE_AUTOUIC OFF)
	set(CMAKE_AUTORCC OFF)
ENDIF()
